CREATE VIEW v_bunga AS  (select `tbl_pinjaman_d`.`id` AS `id`,`tbl_pinjaman_d`.`tgl_bayar` AS `tgl_bayar`,`tbl_pinjaman_d`.`pinjam_id` AS `pinjam_id`,`tbl_pinjaman_d`.`jumlah_bayar` AS `jumlah_bayar`,`tbl_pinjaman_h`.`anggota_id` AS `anggota_id`,`tbl_pinjaman_h`.`jumlah` AS `jumlah`,`tbl_pinjaman_h`.`bunga` AS `bunga`,`tbl_anggota`.`nama` AS `nama`,round(ceiling((((`tbl_pinjaman_h`.`jumlah` / `tbl_pinjaman_h`.`lama_angsuran`) * `tbl_pinjaman_h`.`bunga`) / 100)),-(2)) AS `bunga_pinjaman` from ((`tbl_pinjaman_d` join `tbl_pinjaman_h` on((`tbl_pinjaman_d`.`pinjam_id` = `tbl_pinjaman_h`.`id`))) join `tbl_anggota` on((`tbl_pinjaman_h`.`anggota_id` = `tbl_anggota`.`id`)))) ;
